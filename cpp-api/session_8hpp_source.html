56  session(const implemetation_selector &sel, detail::sdk_c_api callable)
57  : c_api_callable_(callable),

190  void init_accelerator_handle() {
191  mfxIMPL impl;
192  mfxStatus sts = MFXQueryIMPL(session_, &impl);
193  if (sts != MFX_ERR_NONE)
194  return;
195 
196  #ifdef LIBVA_SUPPORT
197  if ((impl & MFX_IMPL_VIA_VAAPI) == MFX_IMPL_VIA_VAAPI) {
198  VADisplay va_dpy = NULL;
199  int fd;
200  // initialize VAAPI context and set session handle (req in Linux)
201  fd = open("/dev/dri/renderD128", O_RDWR);
202  if (fd >= 0) {
203  va_dpy = vaGetDisplayDRM(fd);
204  if (va_dpy) {
205  int major_version = 0, minor_version = 0;
206  if (VA_STATUS_SUCCESS == vaInitialize(va_dpy, &major_version, &minor_version)) {
207  MFXVideoCORE_SetHandle(session_,
208  static_cast<mfxHandleType>(MFX_HANDLE_VA_DISPLAY),
209  va_dpy);
210  }
211  }
212  }
213  accelerator_handle = va_dpy;
214  }
215  #endif
216  }

360  status decode_frame(std::shared_ptr<frame_surface> out_surface,
361  decoder_process_list list = {}) {
362  mfxSyncPoint syncp;
363  mfxFrameSurface1 *surf = NULL;
364 
365  rdr_->get_data(&bits_);
366 
367  mfxBitstream *bts;
368  if (bits_.get_DataLength() == 0 && rdr_->is_EOS()) {
369  bts = nullptr;
370  state_ = state::Draining;
371  }
372  else {
373  bts = bits_();
374  if (auto [buffers, size] = list.get_raw_ext_buffers(); size) {
375  bts->NumExtParam = static_cast<uint16_t>(size);
376  bts->ExtParam = buffers;
377  }
378  else {
379  bts->NumExtParam = 0;
380  bts->ExtParam = nullptr;
381  }
382  }
383 
384  detail::c_api_invoker e({ [](mfxStatus s) {
385  switch (s) {
386  case MFX_ERR_MORE_DATA:
387  return false;
388  case MFX_ERR_MORE_SURFACE:
389  return false;
390  default:
391  break;
392  }
393 
394  bool ret = (s < 0) ? true : false;
395  return ret;
396  } },
397  MFXVideoDECODE_DecodeFrameAsync,
398  session_,
399  bts,
400  nullptr,
401  &surf,
402  &syncp);
403 
404  if (surf) {
405  // out_surface = std::make_shared<frame_surface>(surf);
406  // out_surface.reset(new frame_surface(surf));
407  // std::shared_ptr<frame_surface> tmp = std::make_shared<frame_surface>(surf);
408  // out_surface.swap(tmp);
409  out_surface->inject(surf, out_surface.use_count() + 1);
410  }
411 
412  if (e.sts_ == MFX_ERR_MORE_DATA && state_ == state::Draining) {
413  state_ = state::Done;
414  return status::EndOfStreamReached;
415  }
416  return mfxstatus_to_onevplstatus(e.sts_);

576  status encode_frame(std::shared_ptr<bitstream_as_dst> bs, encoder_process_list list = {}) {
577  status sts;
578  if (!rdr_)
579  throw base_exception("NULL reader ptr", MFX_ERR_NULL_PTR);
580 
581  while (1) {
582  std::shared_ptr<frame_surface> input_surface(nullptr);
583 
584  if (!rdr_->is_EOS()) {
585  input_surface = alloc_input();
586  rdr_->get_data(input_surface);
587  if (rdr_->is_EOS())
588  input_surface.reset();
589  }
590 
591  sts = encode_frame(input_surface, bs, list);
592 
593  switch (sts) {
594  case status::Ok:
595  return status::Ok;
596  case status::EndOfStreamReached:
597  return status::EndOfStreamReached;
598  case status::NotEnoughData:
599  break;
600  case status::NotEnoughBuffer: {
601  bs->realloc();
602  // Assume that frame was cached so we need to increase buffer size only and send new frame to enc
603  break;
604  }
605  default:
606  return sts;
607  }
608  }
609 
610  return sts;
611  }
