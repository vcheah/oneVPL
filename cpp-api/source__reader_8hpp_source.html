52 class raw_frame_file_reader : public frame_source_reader {
53 public:
59  raw_frame_file_reader(uint16_t width,
60  uint16_t heigth,
61  color_format_fourcc format,
62  std::ifstream& ifl)
63  : frame_source_reader(),
64  width_(width),
65  heigth_(heigth),
66  format_(format),
67  ifl_(ifl),
68  eof_(false) {}
69 
71  virtual ~raw_frame_file_reader() {}
72 
76  virtual bool get_data(std::shared_ptr<frame_surface> frame) {
77  auto [info, data] = frame->map(memory_access::write);
78 
80 
81  uint32_t pitch = 0;
82  switch (format_) {
83  case oneapi::vpl::color_format_fourcc::i420: {
84  pitch = data.get_pitch();
85  auto [Y, U, V] = data.get_plane_ptrs_3();
86 
87  // read luminance plane (Y)
88  read_blob(Y, pitch, width_, heigth_);
89 
90  // read chrominance (U, V)
91  read_blob(U, pitch / 2, width_ / 2, heigth_ / 2);
92  read_blob(V, pitch / 2, width_ / 2, heigth_ / 2);
93  break;
94  }
95  case oneapi::vpl::color_format_fourcc::nv12: {
96  pitch = data.get_pitch();
97  auto [Y, UV] = data.get_plane_ptrs_2();
98 
99  // read luminance plane (Y)
100  read_blob(Y, pitch, width_, heigth_);
101 
102  // read chrominance (UV)
103  read_blob(UV, pitch, width_, heigth_ / 2);
104  break;
105  }
106  case oneapi::vpl::color_format_fourcc::bgra: {
107  pitch = data.get_pitch();
108  auto [R, G, B, A] = data.get_plane_ptrs_4();
109 
110  read_blob(B, pitch, width_ * 4, heigth_);
111  }
112  default:
113  throw base_exception("raw_frame_file_reader unsupported format",
114  MFX_ERR_NOT_IMPLEMENTED);
115  }
116  frame->unmap();
117  return !eof_;
118  }
119 
122  bool is_EOS() const {
123  return eof_;
124  }
125 
126 protected:
132  void read_blob(uint8_t* ptr, uint32_t pitch, uint16_t b_width, uint16_t b_height) {
133  for (uint16_t i = 0; i < b_height; i++) {
134  ifl_.read(reinterpret_cast<char*>(ptr + i * pitch), b_width);
135  if (ifl_.gcount() != b_width)
136  eof_ = true;
137  }
138  }
140  uint16_t width_;
142  uint16_t heigth_;
144  color_format_fourcc format_;
146  std::ifstream& ifl_;
148  bool eof_;
149 };

282  bool get_data(bitstream_as_src* bits) {
283  auto lambda = [&](uint8_t* ptr, uint32_t max, bool& eos) {
284  if_.read(reinterpret_cast<char*>(ptr), max);
285  if (if_.eof())
286  eos = true;

288  };
289  bits->pull_in(lambda);
290  return true;
291  }
