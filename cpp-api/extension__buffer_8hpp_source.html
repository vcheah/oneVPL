55 template <typename T, uint32_t ID>
56 class extension_buffer : public extension_buffer_base {
57 protected:
59  class is_extension_buffer {
60  private:
61  typedef char Yes;
62  typedef int No;
63 
64  template <typename C>
65  static Yes& check(decltype(&C::Header));
66  template <typename C>
67  static No& check(...);
68 
69  public:
71  enum { value = sizeof(check<T>(0)) == sizeof(Yes) };
72  };
73 
74 public:
76  template <
77  typename check = typename std::enable_if<is_extension_buffer::value, mfxExtBuffer>::type>
78  extension_buffer() : buffer_() {
79  buffer_.Header.BufferSz = sizeof(T);
80  buffer_.Header.BufferId = ID;
81  }
82 
85  extension_buffer(const extension_buffer& other) = default;
86 
90  extension_buffer& operator=(const extension_buffer& other) = default;
91  virtual ~extension_buffer() {}
92 
93 public:
96  uint32_t get_ID() const {
97  return buffer_.Header.BufferId;
98  }
99 
102  uint32_t get_size() const {
103  return buffer_.Header.BufferSz;
104  }
105 
108  virtual T& get_ref() {
109  return buffer_;
110  }
111 
114  T get() const {
115  return buffer_;
116  }
117 
120  mfxExtBuffer* get_ptr() {
121  return reinterpret_cast<mfxExtBuffer*>(&buffer_);
122  }
123 
124 protected:
126  T buffer_;
127 };

311 #define REGISTER_SINGLE_POINTER_EXT_BUFFER_BEGIN(className, \
312  extBuffer, \
313  bufferID, \
314  ptr, \
315  ptrType, \
316  numElems) \
317  \
318  class className : public extension_buffer_with_ptrs<extBuffer, bufferID> { \
319  public: \
320  \
321  \
322  explicit className(const std::vector<ptrType>& ptr) : extension_buffer_with_ptrs() { \
323  if (ptr.size()) { \
324  buffer_.numElems = (decltype(buffer_.numElems))ptr.size(); \
325  buffer_.ptr = new ptrType[buffer_.numElems]; \
326  std::copy(ptr.begin(), ptr.end(), buffer_.ptr); \
327  } \
328  else { \
329  buffer_.numElems = 0; \
330  buffer_.ptr = NULL; \
331  } \
332  } \
333  \
334  \
335  \
336  className(const className& other) { \
337  this->buffer_ = other.buffer_; \
338  if (other.buffer_.numElems) { \
339  this->buffer_.numElems = other.buffer_.numElems; \
340  this->buffer_.ptr = new ptrType[buffer_.numElems]; \
341  std::copy(other.buffer_.ptr, \
342  other.buffer_.ptr + other.buffer_.numElems, \
343  this->buffer_.ptr); \
344  } \
345  } \
346  \
347  \
348  ~className() { \
349  delete[] buffer_.ptr; \
350  buffer_.numElems = 0; \
351  } \
352  \
353  \
354  \
355  virtual className& operator=(const className& other) { \
356  if (&other == this) \
357  return *this; \
358  \
359  if (this->buffer_.ptr) \
360  delete[] this->buffer_.ptr; \
361  this->buffer_ = other.buffer_; \
362  \
363  if (other.buffer_.numElems) { \
364  this->buffer_.numElems = other.buffer_.numElems; \
365  this->buffer_.ptr = new ptrType[buffer_.numElems]; \
366  std::copy(other.buffer_.ptr, \
367  other.buffer_.ptr + other.buffer_.numElems, \
368  this->buffer_.ptr); \
369  } \
370  else { \
371  buffer_.numElems = 0; \
372  buffer_.ptr = NULL; \
373  } \
374  \
375  return *this; \
376  }
